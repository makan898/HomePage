/* swap関数は、分割コンパイルするバージョン /
/ bcc32 -c 0202.c
bcc32 -e0202 0202.obj swap.obj */

#include “swap.h”
#define N 10

/* 大域変数 /
int a[N];
void print_array(int left, int right);
int partition(int left, int right);
void quick_sort(int left, int right);
void print_array(int from, int to) { / from番目から to番目までの要素をプリント */
int i;

for (i = from ; i <= to; i ++)
printf(“%3d “, a[i]);

printf(“\n\n”);

}

int main(void){
int i;
for(i = 0;i<N;i++){
printf(“a[%d]を入力してください：”,i);
scanf(“%d”,&a[i]);
}
/* 配列a ソート前。表示 */
printf (“BEFORE :”);
print_array(0, N-1);

quick_sort(0, N-1); /* クイックソート! */

/* 配列a ソート後。表示 */
printf (“ AFTER :”);
print_array(0, N-1);

return(0);
}

void quick_sort(int left, int right) {

/*　疑似コードを参考に自作しよう。コードは簡単。納得までねばろう */
int j;
if(left < right){
j = partition(left,right);
quick_sort(left,j - 1);
quick_sort(j + 1,right);
}

}

/* 以下はSTEP2.1 この関数がクイックソートの根幹部です。 */
int partition(int left, int right){
int i, j, pivot;

pivot = a[left]; /* pivotは、最左 /
i = left+1; / 左から右へ動くポインター /
j = right; / 右から左へ動くポインター */

/* 疑似コードを参考に作成しなさい */
while(i <= j){
while(i <= right && a[i] < pivot){
i++;
}
while(a[j] > pivot){
j–;
}
if(i <= j){
swap(&a[i],&a[j]);
i++;
j–;
}
}
swap(&a[left],&a[j]);
return(j);
}
