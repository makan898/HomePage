MaxSample.java は、static な maximum( )メソッドを利用するプログラムである。maximum( )は渡された 1 次元配列の要素の最大値を求め、それを返すメソッドである。MaxSample クラス内で main( )メソッドが記述されているが、maximum( )メソッドの定義は抜けている。maximum( )メソッドの定義を追加し、プログラムを完成せよ。

public class MaxSample {
    // ここに maximum( )メソッドの定義を追加
    static int maximum(int x[]){
        int max = x[1];
        for( int n : x ) {
            if (n > max    ){
                max = n;
            }    
        }
        return max;
    }
    public static void main(String args[ ]){
    int a[ ] = { 10, 20, 50, 15, 5, 40, 35};
    int b[ ] = { -10, -20, -50, -15, -5, -40, -35, -60};
    System.out.println( maximum (a) );
    System.out.println( maximum (b) );
    }
    }

    for (int i = 0; i < args.length; i++) {
    arr[i] = Integer.parseInt(args[i]);
    }
    Array cal = new Array();
    cal.name = "Result";
    cal.add(arr);
    cal.output();
    }
    }
    public class Array{
    String name;
    private int sum = 0;

    public void output(){
    System.out.println(" [ " + name + " ] ");
    System.out.println("sum : " + sum);
    }

    public void add(int[] array){
    for(int i = 0; i < array.length; i++){
    sum = sum + array[i];
    }
    }
    }

  }
boolean sameBlock(Block obj) {
if( (obj.x == x) && (obj.y == y) )return true;
else return false;

}
 // <--sameArea( )メソッドの定義を追加

    if(obj.x * obj.y == x * y)return true;

}
}
